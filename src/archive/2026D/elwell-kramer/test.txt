void main() {
  vec2 p = uv();
  float r = 0.; 
  float g = 0.; 
  float b = 0.; 
  float frequency1 = 5.;
  float frequency2 = 7.;
  float frequency3 = 11.;
  float gain = 1.;
  float thickness = .025;

  for( float i = 0.; i < 1.; i++ ) { 
    p.x += sin( p.y + time * i) * sin(gain); 
    r += abs( thickness / sin(p.x) );
  }
  
  for( float i = 0.; i < 2.; i++ ) { 
    p.x += sin( p.y + time * i) * tan(gain); 
    g += abs( thickness / tan(p.x));
  }
  
  for( float i = 0.; i < 3.; i++ ) { 
    p.x += sin( p.y + time * i) * cos(gain); 
    b += abs( thickness / cos(p.x) );
  }

  gl_FragColor = vec4( r*sin(b)*g,g/b,b*r,1 );
}
